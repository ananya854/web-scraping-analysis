# Web Scraping: US Companies by Revenue # Source: Wikipedia from bs4 import BeautifulSoup import requests import pandas as pd # Step 1: Fetch the page url = 'https://en.wikipedia.org/wiki/List_of_...' page = requests.get(url) soup = BeautifulSoup(page.text, 'html') # Step 2: Find the right table table = soup.find('table', class_='wikitable sortable') world_titles = table.find_all('th') # Step 3: Extract headers world_table_titles = [title.text.strip() for title in world_titles] df = pd.DataFrame(columns=world_table_titles) # Step 4: Extract rows (skip blank first row) column_data = table.find_all('tr') for row in column_data[1:]: row_data = row.find_all('td') individual_row_data = [data.text.strip() for data in row_data] length = len(df) df.loc[length] = individual_row_data # Step 5: Export & Analyze df.to_csv('Companies.csv') df.head() | df.info() | df.describe()
